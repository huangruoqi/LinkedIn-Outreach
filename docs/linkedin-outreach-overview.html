%% MAIN WORKFLOW
%% ─────────────────────────────────

START(["Start Outreach"]) --> DISCOVER

DISCOVER["1  Discover Prospect\nscrape-profile skill\nMCP scrape_profile\nPlaywright extracts profile data\nCreates prospect.json  stage=cold"]

DISCOVER --> ROUTE{"Already\n1st-degree?"}
ROUTE -- "No  2nd/3rd" --> CONNECT
ROUTE -- "Yes" --> ENGAGE

CONNECT["2  Send Connection Request\nsend-connection-request skill\nCompose note  300 char max\nMCP send_connection_request\nPlaywright clicks Connect + Add Note\nPersist  stage=pending_connection"]

CONNECT --> WAIT["3  Wait for Acceptance\nsync-pending-connections skill\nMCP is_first_degree_connection\nPlaywright checks badge"]

WAIT --> ACC{"Accepted?"}
ACC -- "No  over 48h" --> DEAD["Mark Dead\nended_reason = no_response"]
DEAD --> REPORT
ACC -- "Yes" --> ENGAGE

ENGAGE["4  Run Conversation Sequence\nconversation-planner skill\n4 phases per cycle"]

ENGAGE --> SEQ

subgraph CYCLE ["Each Cycle  conversation-planner"]
  direction TB
  CA["Phase A  Sync\nfetch_chat_history\nMerge new messages\nDeduplicate thread"]
  CB["Phase B  Plan\nLoad state via MCP\nState machine evaluates\nCompose next message\nLog PlannedMessage"]
  CC["Phase C  Deliver\nsend_message via MCP\nPlaywright types + sends\nHuman-like timing"]
  CD["Phase D  Persist\nupsert_conversation\nupsert_prospect\nappend_action_log"]
  CA --> CB --> CC --> CD
end

SEQ --> CYCLE
